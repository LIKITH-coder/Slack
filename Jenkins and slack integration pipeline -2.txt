pipeline {
   agent any
   stages {
      stage{
         steps{
           echo 'slack notification'
         }
      }
   }
   
    post {
        always {
            script {
                slackSend(
                    channel: '#general',
                    message: "please check deatils in '${env.JOB_NAME} ${env.BUILD_NUMBER} ${env.BUILD_URL}'"
                )
            }
        }
    }
}
