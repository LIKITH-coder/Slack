jenkins and slack integration in jenkins pipeline -1


def COLOR_MAP = [
   SUCCESS: 'good', 
   UNSTABLE: 'warning', 
   FAILURE: 'danger'
]

pipeline {
   agent any
   stages{
      stage(git) {
          steps{
              git 
          }
      }
   post {
        always {
            script {
                slackSend (
                    channel: '#jenkins-ci',
                    color: COLOR_MAP[currentBuild.currentResult],
                    message: "please check deatils in '${env.JOB_NAME} ${env.BUILD_NUMBER} ${env.BUILD_URL}'"
                )
           }
       }
   }
}
}
